```css
body {
  cursor: url(https://cdn.jsdelivr.net/gh/Erica-Iris/Blog_/Arrow.cur),
    default;
}

  cursor: url(https://cdn.jsdelivr.net/gh/Erica-Iris/Blog_/IBeam.cur),
    text;
}
a.hover {
  cursor: url(https://cdn.jsdelivr.net/gh/Erica-Iris/Blog_/hand.cur),
    pointer;
}
p:hover {
  cursor: url(https://cdn.jsdelivr.net/gh/Erica-Iris/Blog_/IBeam.cur),
    text;
}
span:hover {
  cursor: url(https://cdn.jsdelivr.net/gh/Erica-Iris/Blog_/IBeam.cur),
    text;
}
text.hover {
  cursor: url(https://cdn.jsdelivr.net/gh/Erica-Iris/Blog_/IBeam.cur);
}
```
